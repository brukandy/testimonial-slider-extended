<script>
// Auto-resize iframe based on content
window.addEventListener('message', function(e) {
    if (e.data.testimonialHeight) {
        const iframe = document.querySelector('iframe[src*="testimonial-slider-extended"]');
        if (iframe) {
            iframe.style.height = e.data.testimonialHeight + 'px';
        }
    }
    
    // Lock/unlock scroll when modal opens/closes (SAFE version)
    if (e.data.action === 'lockScroll') {
        // Solo se non è già bloccato da qualcos'altro
        if (!document.body.dataset.scrollLocked) {
            document.body.dataset.scrollLocked = 'true';
            document.body.dataset.scrollPosition = window.pageYOffset;
            document.body.style.overflow = 'hidden';
            document.body.style.position = 'fixed';
            document.body.style.width = '100%';
            document.body.style.top = `-${window.pageYOffset}px`;
        }
    }
    
    if (e.data.action === 'unlockScroll') {
        // Solo se l'abbiamo bloccato noi
        if (document.body.dataset.scrollLocked === 'true') {
            const scrollY = document.body.dataset.scrollPosition || 0;
            document.body.style.overflow = '';
            document.body.style.position = '';
            document.body.style.width = '';
            document.body.style.top = '';
            delete document.body.dataset.scrollLocked;
            delete document.body.dataset.scrollPosition;
            window.scrollTo(0, parseInt(scrollY));
        }
    }
});
</script>
